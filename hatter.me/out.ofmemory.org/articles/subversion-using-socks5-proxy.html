作者 Hatter Jiang | 发布于 2014-02-06

http://www.pps.univ-paris-diderot.fr/~jch/software/polipo/

-rwxr-xr-x polipo

socksParentProxy = "localhost:7777"
socksProxyType = socks5
proxyPort = 7778

polipo -c c.conf > log 2>&1 &
