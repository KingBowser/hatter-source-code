    // 代码1
public class Sample {
  private static int count = 0;

    count++;
  }
}
  
// 代码2
public class Sample {
  private static AtomicInteger count = new AtomicInteger(0);
  
  public static void increment() {
    count.getAndIncrement();
  }
}
